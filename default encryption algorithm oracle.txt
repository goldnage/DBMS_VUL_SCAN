1. check default tablespace algorithm
SHOW PARAMETER TABLESPACE_ENCRYPTION_DEFAULT_ALGORITHM;

2. view active encryption for existing Tablespaces
SELECT TS#, ENCRYPTIONALG, STATUS 
FROM V$ENCRYPTED_TABLESPACES;

3. identify default column encryption
SELECT OWNER, TABLE_NAME, COLUMN_NAME, ENCRYPTION_ALG 
FROM DBA_ENCRYPTED_COLUMNS;

4. databse-level key information
SELECT ENCRYPTIONALG FROM V$DATABASE_KEY_INFO;
